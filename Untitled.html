<body>

  <div class="portfolio" id="portfolio"></div>


  <script>

    // Editable JSON with your photos and videos

    const media = [

      { type: "image", src: "https://via.placeholder.com/600x400", caption: "Photo 1" },

      { type: "video", src: "https://www.w3schools.com/html/mov_bbb.mp4", caption: "Video 1" },

      { type: "image", src: "https://via.placeholder.com/600x400/0000FF", caption: "Photo 2" }

    ];


    const portfolio = document.getElementById("portfolio");


    media.forEach(item => {

      const card = document.createElement("div");

      card.className = "item";

      

      if (item.type === "image") {

        card.innerHTML = `<img src="${item.src}" alt="">`;

      } else if (item.type === "video") {

        card.innerHTML = `<video controls>

                            <source src="${item.src}" type="video/mp4">

                          </video>`;

      }

      

      if (item.caption) {

        card.innerHTML += `<div class="caption">${item.caption}</div>`;

      }

      

      portfolio.appendChild(card);

    });

  </script>
